Adding extra profiling to the Caveman2Cosmos project

The following script was used to analyse and add extra profiling to around 2000 methods (out of 17 000) in the Caveman2Cosmos project.

The script is verbose for possible human interaction, but should be readable.

110# adds #include "incname" in the file (either path or SourceFile) if it is not present
111def addInclude(file, incname):
112 if isinstance(file, str):
113 try:
114 src = [ src for src in sources
115 if src.filepath() == file or crawler.sourceRelPath(src) == file ][0]
116 except IndexError:
117 warning(f"Cannot add #include, file not on the source list: {crawler.sourceRelPath(file)}")
118 return
119 else:
120 src = file
121
122 matchPref = src.matchUnscoped(srcPrefixPat)
123 qIncluded = src.findUnscoped(r'#include[ \t]+"(?i:'f"{incname}"r')"')
124 aIncluded = src.findUnscoped(r'#include[ \t]+<(?i:'f"{incname}"r')>')
125 included = qIncluded or aIncluded
126
127 if not matchPref:
128 if included:
129 includesWarn.setdefault(src, []).append(("bad prefix", incname))
130 warning(f"Included, but cannot parse the file prefix: {incname}", src)
131 return
132 else:
133 error(f"Not included and cannot parse the file prefix: {incname}", src)
134
135 if src.filepath().endswith(".h") and not matchPref.group("prag") and not matchPref.group("ifdef"):
136 warning(f"Header not guarded", src)
137
138 if incname.lower() in map(str.lower, matchPref.captures("inc1")):
139 info(f"Already included: {incname}", src)
140
141 elif incname.lower() in map(str.lower, matchPref.captures("inc2")):
142 includesWarn.setdefault(src, []).append(("in <>", incname))
143 warning(f"Included, but in <> brackets: {incname}", src)
144
145 elif included:
146 includesWarn.setdefault(src, []).append(("not in prefix", incname))
147 warning(f"Included, but not present in the expected file prefix: {incname}", src)
148
149 else:
150 if matchPref.group("inc1"): src.insert(matchPref.starts("insert2")[0], f'#include "{incname}"\r\n')
151 elif matchPref.group("inc2"): src.insert(matchPref.starts("insert3")[0], f'#include "{incname}"\r\n')
152 else: src.insert(matchPref.start("insert1"),
153 f'\r\n#include "{incname}"\r\n' + ("" if matchPref.group("insert1") else "\r\n"))
154 includesToAdd.setdefault(src, []).append(incname)
155 modifiedSrcs.add(src)
156 info(f"Include directive added: {incname}", src)

160#### Code qualification for profiling ####
161
162# macros recognized as a profiling start
163profFuncRe = r"\bPROFILE_(?P<newprof>EXTRA_)?FUNC\b|\bPROFILE(?:_BEGIN|_THREAD\b|\b)"
164# all macros that are considered a method header (profiled methods are expected to call profiling in the header)
165hdMacroRe = r"(?P<mac>PROXY_TRACK|(?P<prof>"f"{profFuncRe}"r"))[^;\n]*(?:;|\r\n)"
166methBlockPat = regex.compile(
167 r"\{"
168 f"(?:{__}|{hdMacroRe})*+"
169 r"(?P<body>(?s:.*?))"
170 f"{___}"
171 r"}")
172profPat = regex.compile(r"\bPROFILE")
173loopPat = regex.compile(r"\bwhile\b|\bfor\b|(?i:foreach\w*\s*\‍()")
174
175
176# qualifies a method block for profiling (positive if it contains a loop or a call that is likely an iteration)
177# begin - start position of a block (the { character),
178# prot - parsed method prototype,
179# ln - line number where the prototype begins
180def qualifyBlock(src, begin, prot, ln):
181 end = src.blockEnd(begin)
182 code = src.intCode(begin, end + 1)
183 match1 = src.fullmatchUnscoped(methBlockPat, begin, end + 1)
184 if not match1:
185 warning(f"Could not match the method's body: {prot}", src, ln)
186 return False
187
188 bodyBegin = match1.start("body")
189 bodyEnd = match1.end("body")
190
191 if match1.group("newprof"):
192 alreadyProfiledNew.setdefault(src, []).append((ln, prot, code, begin))
193 info(f"Already profiled (extra): {prot}", src, ln)
194 return False
195
196 if match1.group("prof"):
197 alreadyProfiledStd.setdefault(src, []).append((ln, prot, code, begin))
198 info(f"Already profiled (standard): {prot}", src, ln)
199 return False
200
201 if src.findUnscoped(profPat, bodyBegin, bodyEnd):
202 uncertainProfiling.setdefault(src, []).append((ln, prot, code, begin))
203 warning(f"Pofile macros that cannot be qualified in: {prot} (SKIPPING)", src, ln)
204 return False
205
206 methname = Syntax.extractMethProtId(prot)[0][-1]
207 if ( (match2 := src.findUnscoped(r"(?:\b(\w+)\s*->\s*)?"f"\b{methname}"r"\s*\‍(", bodyBegin, bodyEnd)) and
208 match2.group(1) == "this"):
209
210 unprofiledMaybeRec.setdefault(src, []).append((ln, prot, code, begin))
211 warning(f"Possibly recursive unprofiled method: {prot} (SKIPPING)", src, ln)
212 return False
213
214 if src.findUnscoped(loopPat, bodyBegin, bodyEnd):
215 profileToAdd.setdefault(src, []).append((ln, prot, code, begin))
216 info(f"Adding profiling to: {prot}", src, ln)
217 return True
218
219 else:
220 profilingNotQual.setdefault(src, []).append((ln, prot, code, begin))
221 return False

225#### Files processing ####
226
227# regular expression used to find method/function definitions (sufficient in this project)
228methFinderPat = SourceFile.makeSkipBlocksPat(r"(?:\b\w+"f"{___}::{___})*{Syntax.methFinderRe}")
229
230initRe = r"(?::(?:\s|"f"{Syntax.clipRe}"r"|[^{])*+)"
231methProtSuffixPat = regex.compile(f"{___}{initRe}?{___}"r"(?:(?P<block>\{)|[;=])") # possible suffixes of
232 # a method prototype
233
234
235# following functions take a SourceFile and set search scopes in it
236scopeSelectors = [lambda src: True, # leavs the default (total) scope
237 lambda src: src.tryScopeToNamespaceBody(r"\w+", skipBlocks=False), # scope to all namespaces
238 lambda src: src.tryScopeToClassBody("*", r"\w+", skipBlocks=False)] # scope to all classes

240# reads sources and qualifies methods for profiling
241def qualify():
242 qualified = 0
243
244 for src in sources:
245 if qualified == limit: break
246
247 if crawler.sourceRelPath(src).lower() in excludedFiles:
248 warning(f"File excluded", src)
249 skippedFiles.add(src)
250 continue
251
252 methBlockBegins = []
253
254 for selectScope in scopeSelectors:
255 if not selectScope(src): continue
256
257 for (match1, scopeTag) in src.findAll(methFinderPat, skipBlocks=True, scopeTag=True):
258 ln = match1.startLoc()[0]
259 match Syntax.parseMethPrototype(src.intCode(), match1.intStart()):
260 case (prot, _, pos):
261 pos = src.orgPos(pos)
262 match scopeTag:
263 case (_, name): prot = f"{name}::{prot}" # in this case the scope is a class
264 match2 = src.matchUnscoped(methProtSuffixPat, pos)
265 if not match2:
266 unparsable1.setdefault(src, []).append((ln, prot))
267 continue
268 if match2.group("block"):
269 if prot in excludedProts:
270 warning(f"Prototype excluded: {prot}", src, ln)
271 skippedDefs.setdefault(src, []).append((ln, prot))
272 continue
273 methBlockBegins.append((match2.start("block"), prot, ln))
274
275 case None:
276 unparsable0.setdefault(src, []).append((ln, match1.group(0)))
277
278 src.resetScopes()
279
280 for (begin, prot, ln) in methBlockBegins:
281 if qualifyBlock(src, begin, prot, ln): qualified += 1
282 if qualified == limit:
283 break

303# adds profiling macros to methods stored in profileToAdd
304def addProfiling():
305 addFProfilerTo = set() # list of source files where to include FProfiler.h
306
307 for src in profileToAdd:
308 replMatches = [] # list of pairs (SourceMatch, code), where the former is a 0-width match that points the
309 # place to insert and the latter is the profiling code to be inserted - later used as a list
310 # of tagged matches
311
312 for (ln, prot, code, begin) in profileToAdd[src]:
313 (pos, space) = src.posToInsertBlockSPrefix(begin) # position to insert a single-line block prefix and
314 # space to be added behind for indentation preservation
315 replMatches.append((SourceRangeMatch(src, (pos, pos), copySource=False),
316 f"{funcExtraProfileMacro}();" + space))
317
318 if replMatches:
319 src.replaceMatches(replMatches, lambda tmatch: tmatch[1]) # the method takes a list of possibly tagged
320 # matches and a function that for such a match
321 # returns the replacement code
322 modifiedSrcs.add(src)
323 addFProfilerTo.add(src.filepath())
324
325 print()
326 print("PROFILE MACROS ADDED")
327 print()
328
329 # the following including procedure is sufficient in this project (in general some .cpp could miss FProfiler.h)
330 for filepath in addFProfilerTo:
331 if not (filepath.endswith(".cpp") and filepath[:-3] + "h" in addFProfilerTo):
332 addInclude(filepath, "FProfiler.h")
333
334 print()
335 print("---")
336 print("Profile macros added (variable: profileToAdd):", len(flatten(profileToAdd.values())))
337 print("Include directives added (variable: includesToAdd):", len(flatten(includesToAdd.values())))
