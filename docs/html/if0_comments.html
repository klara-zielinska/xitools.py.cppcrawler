Replacing #if 0 blocks with comments

The following script comments out all #if 0 blocks in the source files in a Sources directory. It does not support #elif, but it is not hard to add it – we will leave it as an exercise to not complicate the example (note that #elif 0 should also be taken into consideration).

02_comment_if0s\comment_if0s.py
1from xitools.cppcrawler import *
2crawler = CppCrawler("Sources", backupDir="Backup", encoding="utf-8")
3
4def srcMatchMsg(src, match, msg):
5 return f"{src.filepath()}:{match.startLoc()[0]}: {msg}"
6
7# given an #if... directive position, finds the corresponding #else, #elif or #endif (whichever comes first)
8def findEndifElse(src, pos, *, findElse=True):
9 elseRe = r"|else|elif" if findElse else ""
10 preprocRe = r"(?<=^|\n)[ \t]*#(?P<dir>if|endif"f"{elseRe}).*\n?" # matches #ifdef, #ifndef
11 while match := src.findUnscoped(preprocRe, pos):
12 if match.group("dir") == "if":
13 match2 = findEndif(src, match.end())
14 assert match2, srcMatchMsg(src, match, "#if has no #endif")
15 pos = match2.end()
16 else:
17 return match
18 return None
19
20def findEndif(src, pos):
21 return findEndifElse(src, pos, findElse=False)
22
23toSave = set()
24
25for src in crawler.loadSourceFiles(["*.h", "*.cpp"]):
26 pos = 0
27 while match := src.findUnscoped(r"(?<=^|\n)[ \t]*#if[ \t]+0\b.*\n?", pos):
28 match2 = findEndifElse(src, match.end())
29 assert match2, srcMatchMsg(src, match, "#if has no #endif")
30 pos = match2.end()
31
32 match match2.group("dir"):
33 case "elif":
34 print(srcMatchMsg(src, match2, "#elif not supported"))
35 continue
36
37 case "else":
38 match3 = findEndif(src, match2.end())
39 assert match3, srcMatchMsg(src, match, "#if has no #endif")
40 src.insert(match3.start(), "//") # code modifications invalidate matched positions behind, thus
41 # we need to perform them latter first
42
43 src.setScope(match.start(), match2.start())
44 src.replace(r"(?<=^|\n)", "//")
45 toSave.add(src)
46 # after inserting // pos can be inside a comment - this is ok, because findUnscoped omits comments
47
48crawler.saveSources(toSave)
49
50print("SCRIPT FINISHED")
